## üß™ –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è:

```python
data = {
    'Name': ['–ê–Ω–Ω–∞', '–ë–æ—Ä–∏—Å', '–ò—Ä–∏–Ω–∞'],
    'Age': [25, 30, 28],
    'City': ['–ú–æ—Å–∫–≤–∞', '–°–ü–±', '–ö–∞–∑–∞–Ω—å']
}

print(df.describe())     # —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –ø–æ Age

```
